\title{Information Erasure Inside the Quantum of Action}

\begin{abstract}
Quantum microstates are operationally indistinguishable inside Planck's quantum of action. We show that Landauer's principle compels the erasure of each microstate into the minimal-action state. This erasure granulates the continuum, selecting the discrete configurations of quantum mechanics at low action and smooth classical trajectories at high action.
\end{abstract}

\label{eq:CovarianceEllipsoid}

\label{eq:EllipsoidArea}

Saturating the Heisenberg bound ($\sigma_q \sigma_p = \hbar/2$) gives the minimum area:

\begin{equation}
\label{eq:SymplecticCapacity}
A_{\text{blob}} = 2\pi \left( \frac{\hbar}{2} \right) = \frac{h}{2}
\end{equation}

\subsection{One Blob or Two?}

Zurek \cite{Zurek2001} demonstrated that a quantum state constrained by the classical action boundaries $\Delta q$ and $\Delta p$ develops sub-Planck structures defined by the reciprocal limits:

\begin{equation}
\label{eq:BlobSqueeze}
\delta q = \frac{\hbar}{\Delta p} \quad \text{and} \quad \delta p = \frac{\hbar}{\Delta q}
\end{equation}

Expressing Zurek's reciprocal limits as phase-space areas yields a conjugate pair of de Gosson capacities (see Fig.~\ref{fig:erase_schematic}):

\begin{equation}
\label{eq:BlobAreas}
\pi \delta q \Delta p = \frac{h}{2} \quad \text{and} \quad \pi \Delta q \delta p = \frac{h}{2}
\end{equation}

Because both areas are required to bound the space, the total quantum of action $h$ is defined by their geometric sum:

\begin{equation}
\label{eq:SimpleLanguageBlobs}
\text{Quantum of Action} = \text{Pair of Quantum Blobs}
\end{equation}

\begin{equation}
\label{eq:Blobs}
h = \pi \delta q \Delta p + \pi \Delta q \delta p
\end{equation}

Planck demonstrated that a classical phase space of area $A$ has a capacity of $N = A/h$ discrete quanta. Scaling the local quantum boundaries ($\delta q, \delta p$) by this macroscopic capacity, we construct the upper limit of the Erasure Action:

\begin{equation}
\label{eq:GeometricActionLimit}
K E_L \tau \le \frac{A}{h} (\pi \delta q  \Delta p + \pi \Delta q \delta p)
\end{equation}

Because the minimal-action erasure path from a microstate to its target coordinate leaves residual displacements $|\epsilon_q| \le \delta q$ and $|\epsilon_p| \le \delta p$, the exact Erasure Action is the scaled area of these physical residuals:

\begin{equation}
\label{eq:ExactGeometricAction}
K E_L \tau = \frac{A}{h} (\pi |\epsilon_q| \Delta p + \pi \Delta q |\epsilon_p|)
\end{equation}

Because Zurek's reciprocal limits established $\pi \Delta p = \frac{h}{2\delta q}$ and $\pi \Delta q = \frac{h}{2\delta p}$:

\begin{equation}
\label{eq:DimensionlessSubstitution}
K E_L \tau = \frac{A}{h} \left( |\epsilon_q| \frac{h}{2\delta q} + |\epsilon_p| \frac{h}{2\delta p} \right) = \frac{A}{2} \left( \frac{|\epsilon_q|}{\delta q} + \frac{|\epsilon_p|}{\delta p} \right)
\end{equation}

\begin{equation}
\label{eq:Saturated}
K E_L \tau = \frac{A}{2} \left( \frac{\delta q}{\delta q} + \frac{\delta p}{\delta p} \right) \implies K E_L \tau \le \frac{A}{2} (1+1)
\end{equation}

\begin{equation}
\label{eq:HeroDerivation}
K E_L \tau \le A
\end{equation}

L(q,p) = \sum_{\mu=1}^{N} \mathbb{I}\left( \frac{A}{h} \epsilon_q = q \right) \mathbb{I}\left( \frac{A}{h} \epsilon_p = p \right)

\begin{equation}
\label{eq:Marginals}
N(q) = \sum_{p} L(q,p)
\quad \text{and} \quad
N(p) = \sum_{q} L(q,p)
\end{equation}

Let the generic variable $x \in \{q, p\}$ denote any phase-space component.

\begin{algorithmic}[1]
\Function{ENCODE}{$x_B, \delta x$}
    \State $\text{left} \gets (-1, 0)$
    \State $\text{num} \gets 0, \quad \text{den} \gets 1$
    \State $x_\mu \gets \text{num} / \text{den}$
    \State $\text{right} \gets (1, 0)$
    \State $s_x \gets \text{``''}$
    \While{$|x_\mu - x_B| > \delta x$}
        \If{$x_\mu < x_B$} 
            \State $\text{left} \gets (\text{num}, \text{den})$
            \State $s_x \gets s_x + \text{``1''}$
        \Else 
            \State $\text{right} \gets (\text{num}, \text{den})$
            \State $s_x \gets s_x + \text{``0''}$
        \EndIf
        \State $\text{num} \gets \text{left}_{\text{num}} + \text{right}_{\text{num}}$
        \State $\text{den} \gets \text{left}_{\text{den}} + \text{right}_{\text{den}}$
        \State $x_\mu \gets \text{num} / \text{den}$
    \EndWhile
    \State $\epsilon_x \gets x_\mu - x_B$
    \State \Return $\epsilon_x, x_\mu, s_x$
\EndFunction
\end{algorithmic}

We invert and reverse the encoding sequence $s_x$ to form the minimal erasure path, $s_x^* = \operatorname{reverse}(\overline{s_x})$, which is irreversibly destroyed.

\begin{algorithmic}[1]
\Function{ERASE}{$s_x^*$}
    \State $K_x \gets |s_x^*|$
    \While{$K_x > 0$}
        \State \Call{ShiftLeft}{$s_x^*$} 
        \Statex \Comment{\text{Dissipate Landauer heat} $k_B T \ln{2}$ \text{over time} $\tau$}
        \State $K_x \gets |s_x^*|$
    \EndWhile
\EndFunction
\end{algorithmic}
